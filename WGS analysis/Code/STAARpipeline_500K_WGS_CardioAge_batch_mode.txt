############ Single variant analysis
# Define the number of sub-tasks for each chromosome
Single_Array=(34 33 27 26 25 23 22 20 19 18 19 18 14 13 12 13 12 11  8  9  6  6)
# 388 jobs

# Loop through all chromosomes
for CHR in {1..22}
do
  # Get the number of sub-tasks for the current chromosome
  num_arrays=${Single_Array[$((CHR-1))]}
  
  # Loop through all sub-tasks for the current chromosome
  for i in $(seq 1 $num_arrays)
  do
	
    dx run staarpipeline --priority=low \
      -inullobj_file=${nullmodel_WGS} \
      -iagds_file=ukbb_lilab:/UKB_500K_WGS_AGDS_uncompressed_newQClabel_HWE/ukb.500k.wgs.chr${CHR}.pass.annotated.gds \
      -itest_type=Single \
      -ivariant_type=variant \
      -imin_mac=10 \
      -iuser_cores=15 \
      -iarrayid=${i} \
      -ioutfile=UKBB_WGS_CardioAge_Single_Variant_chr${CHR}_arrayid${i} \
      -iqc_label_dir=annotation/info/QC_label3 \
	  -isubset_variants_num=5000 \
      --destination=${path_Single} --instance-type="mem3_ssd1_v2_x16" --yes
  done
done

# If a job fails after 10 attempts, it needs to be resubmitted (e.g., chr1_arrayid1).
CHR_LIST=(1 1)
ARRAYID_LIST=(1 2)

for ((i=0; i<${#CHR_LIST[@]}; i++)); do

  	dx run staarpipeline --priority=low \

      -itest_type=Single \
      -ivariant_type=variant \
      -imin_mac=10 \
      -iuser_cores=15 \

      -ioutfile=UKBB_WGS_CardioAge_single_variant_chr${CHR}_arrayid${arrayid} \
      -iqc_label_dir=annotation/info/QC_label3 \
      -isubset_variants_num=5000 \

done

results_single_batch <- c()

for(chr in 1:22){
  print(paste0("Chr: ",chr))
  
  # Loop through all sub-tasks for the current chromosome
  for(i in seq(Single_Array[chr])){
    print(paste0("i: ",i))
    results_single <- get(load(paste0("/mnt/project",input_path,Single_results_name,chr,"_arrayid",i,".Rdata")))

    results_single_batch <- c(results_single_batch,results_single)
  }
}

# Save the merged results to an RData file
save(results_single_batch,file="results_single_batch.Rdata")
system(paste0("dx upload results_single_batch.Rdata --path ",paste0(output_path,"results_single_batch.Rdata")))
